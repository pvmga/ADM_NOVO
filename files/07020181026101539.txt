UPDATE C SET C.CADASTRO_DATA =
          (SELECT MIN(AU.DATA_HORA)
            FROM AUDIT_ITENS AU
            WHERE AU.COD_ITEM = C.CODIGO)
FROM CAD_IEST C
WHERE
    C.CADASTRO_DATA IS NULL

GO
UPDATE CAD_IEST SET DATA_ULTIMA_SAIDA = '';
GO

UPDATE C SET C.DATA_ULTIMA_SAIDA =
  (SELECT MAX(V.DATA_FECHAMENTO)
    FROM VENDA V
    JOIN ITENSVEN I ON(I.COD_VENDA = V.COD_VENDA)
    WHERE
    	I.COD_PROD = C.CODIGO)
FROM CAD_IEST C
